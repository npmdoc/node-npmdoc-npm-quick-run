    function runPrefix(prefix) {
  const pkg = loadJson()

  if (prefix === '-i') {
    inquireScript(pkg)
      .then(result => runScript(result.script, pkg))
    return
  }

  if (!pkg.scripts) {
    console.error('Cannot find any scripts in the current package')
    process.exit(-1)
  }

  if (!prefix) {
    printAllScripts(pkg)
    return
  }

  runScript(prefix, pkg)
}
